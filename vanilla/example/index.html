Getting Started

Choose your preferred installation method and start building with Editium

Install via Node Package Manager

npm install editium

Install via Yarn package manager

yarn add editium

Install via PNPM package manager

pnpm add editium

Single file includes CSS, icons, and JavaScript

unpkg

<script src="https://unpkg.com/editium/vanilla/editium.bundle.js"></script>

jsDelivr

<script src="https://cdn.jsdelivr.net/npm/editium/vanilla/editium.bundle.js"></script>

For custom configurations

unpkg

<link rel="stylesheet" href="https://unpkg.com/editium/vanilla/editium.css">
<script src="https://unpkg.com/editium/vanilla/editium.js"></script>

jsDelivr

<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/editium/vanilla/editium.css">
<script src="https://cdn.jsdelivr.net/npm/editium/vanilla/editium.js"></script> 
import { Editium } from 'editium';

function App() {
  return (
    <Editium
      initialValue={[]}
      toolbar="all"
      placeholder="Start typing..."
      height="300px"
      minHeight="200px"
      maxHeight="400px"
      showWordCount={true}
      onChange={(html, json) => {
        console.log(html, json);
      }}
    />
  );
}
const editor = new Editium({
  container: document.getElementById('editor'),
  placeholder: 'Start typing...',
  toolbar: 'all',
  height: '300px',
  minHeight: '200px',
  maxHeight: '400px',
  showWordCount: true,
  onChange: (data) => {
    console.log(data.html, data.json);
  }
});
